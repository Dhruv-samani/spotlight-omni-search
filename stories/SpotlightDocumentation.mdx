# Spotlight Omni Search

A professional, **Tailwind-Native** Spotlight Search component for React and Next.js.

## ðŸš€ Quick Start

<div className="subheading">Next.js App Router</div>

<Source code={`
// app/layout.tsx
"use client";
import { SpotlightProvider, SearchTrigger } from "spotlight-omni-search/next";
import { useRouter } from "next/navigation";

export default function RootLayout({ children }) {
  const router = useRouter();

const items = [
{ id: "home", label: "Home", route: "/" },
{ id: "docs", label: "Documentation", route: "/docs" },
];

return (

<html lang="en">
<body>
<SpotlightProvider
items={items}
onNavigate={(path) => router.push(path)} >
<SearchTrigger />
{children}
</SpotlightProvider>
</body>
</html>
);
}
`} language="tsx" />

<div className="subheading">Vite / React</div>

<Source code={`
import { SpotlightProvider, SearchTrigger } from "spotlight-omni-search/react";
import { useNavigate } from "react-router-dom";

function App() {
const navigate = useNavigate();
const items = [{ id: "dashboard", label: "Dashboard", route: "/dashboard" }];

return (

<SpotlightProvider items={items} onNavigate={navigate}>
<div className="p-4">
<SearchTrigger />
</div>
</SpotlightProvider>
);
}
`} language="tsx" />
